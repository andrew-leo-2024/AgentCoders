# controller_prototype.py
import time, uuid, json
from datetime import datetime, timedelta

# --- placeholders for your GovernanceFactory SDKs (already implemented) ---
from governance_factories.github import GitHubGovernanceFactory
from governance_factories.azure import AzureDevOpsGovernanceFactory
from vault import VaultClient
from audit import AuditLog

# Initialize singletons (controller-only)
vault = VaultClient()  # holds long-lived provider creds
github_gov = GitHubGovernanceFactory(vault.get('github_app_credentials'))
azure_gov  = AzureDevOpsGovernanceFactory(vault.get('azure_sp_credentials'))
audit = AuditLog()

# ----- utility helpers -----
def new_request_id():
    return str(uuid.uuid4())

def now_iso():
    return datetime.utcnow().isoformat() + "Z"

# ----- ephemeral token minting through GitHubGovernanceFactory (factory mints narrow token) -----
def mint_github_ephemeral(role_template, org, reason, ttl_minutes=30):
    req_id = new_request_id()
    # Controller calls the governance factory to mint a scoped token for the agent
    token = github_gov.mint_token(
        role_template=role_template,
        org=org,
        reason=reason,
        ttl_minutes=ttl_minutes,
        request_id=req_id
    )
    audit.record({
        "request_id": req_id,
        "provider":"github",
        "action":"mint_token",
        "role_template":role_template,
        "org":org,
        "ttl":ttl_minutes,
        "timestamp":now_iso(),
        "reason":reason
    })
    return token, req_id

# ----- GitHub bootstrap flow: repo, team, CI, branch protection, secrets -----
def bootstrap_github_repo_and_team(org, repo_name, founder_email):
    # 1. mint token for RepoCreator role
    token, req_id = mint_github_ephemeral(role_template="dev", org=org, reason=f"bootstrap {repo_name}", ttl_minutes=45)
    # 2. call governance factory (agent runtime would use token)
    # but controller can also perform the orchestration by providing the token to worker or doing the calls server-side
    repo = github_gov.create_repository(
        token=token,
        org=org,
        name=repo_name,
        private=True,
        description="Tenant repo created by AI-controller"
    )
    audit.record({"request_id": req_id, "action":"create_repository", "org":org, "repo":repo_name, "result": repo})

    team = github_gov.create_team(token=token, org=org, name=f"{repo_name}-devs", privacy="closed")
    github_gov.add_member_to_team(token=token, org=org, team_slug=team['slug'], member_email=founder_email)
    audit.record({"request_id": req_id, "action":"create_team", "team":team})

    # push initial scaffold commit via the factory
    scaffold_pr = github_gov.create_initial_commit_from_template(
        token=token,
        org=org,
        repo=repo_name,
        template_repo="platform/tenant-template",
        initial_files={"README.md":"# tenant repo"}
    )
    audit.record({"request_id": req_id, "action":"initial_commit", "pr":scaffold_pr})

    # create Actions workflow (CI)
    workflow_yml = """
    name: CI
    on: [push, pull_request]
    jobs:
      build:
        runs-on: ubuntu-latest
        steps:
         - uses: actions/checkout@v4
         - name: Run tests
           run: echo "run tests here"
    """
    github_gov.create_workflow_file(token=token, org=org, repo=repo_name, path=".github/workflows/ci.yml", content=workflow_yml)
    audit.record({"request_id": req_id, "action":"create_workflow", "repo":repo_name})

    # branch protection
    github_gov.create_branch_protection(token=token, org=org, repo=repo_name, branch="main", require_pr=True, required_reviewers=1)
    audit.record({"request_id": req_id, "action":"branch_protection", "repo":repo_name})

    # set secrets via factory (factory maps to GitHub secrets API)
    github_gov.create_secret(token=token, org=org, repo=repo_name, name="TENANT_API_KEY", value="SECRET_PLACEHOLDER")
    audit.record({"request_id": req_id, "action":"create_secret", "repo":repo_name})

    # revoke ephemeral token early (best practice)

# Controller — Documentation & Runnable POC

> **Purpose (one-liner):** The Controller is the trusted control-plane that validates policy, mints ephemeral tokens for Workers, enforces approval gates, and writes signed, append-only audit logs. It never exposes long-lived provider credentials directly to Workers.

## 1. Purpose & scope

This Controller prototype is a minimal, production-minded control-plane for an autonomous-agent platform that uses your GovernanceFactory wrappers (GitHubGovernanceFactory, AzureDevOpsGovernanceFactory, etc.).

**What this doc covers:**

* Token minting and TTL management
* Approval gating for risky ops
* Audit logging (immutable, signed)
* Controller API design for Workers
* Runnable POC instructions and a minimal Flask example

**What this doc does NOT cover:**

* The GovernanceFactory internals (those are separate wrappers)
* Full UI/UX for approval flows (API + webhook hooks supported)

## 2. High-level architecture

```
  +----------+         +--------------------+         +-------------------+
  | Operator | <-----> |     Controller     | <-----> | GovernanceFactory |
  |  (human) |  HTTP   | (policy & vault)   |  SDK    |   (github / azure) |
  +----------+         +--------+-----------+         +--------+----------+
                                  |                               |
                                  | ephemeral tokens             |
                                  v                               v
                            +-----+------+                 +------+------+
                            |   Worker   |                 |   Worker    |
                            | (RepoCreator)               | (CIWriter)   |
                            +------------+                 +-------------+

```

Key points:

* Controller stores long-lived provider creds in a Vault (e.g., HashiCorp Vault or cloud KMS). Vault is only accessible to Controller.
* Controller exposes a narrow REST API for Workers to request ephemeral tokens and to submit audit events. Workers only call GovernanceFactory **via the provided token**.
* All actions are recorded to an append-only, tamper-evident audit store.

---

## 3. Responsibilities

**Controller** must:

* Validate incoming requests (naming rules, quotas, policy templates).
* Return a plan/dry-run for significant operations.
* Gate `approval_required` operations via the Approval API (webhook to operator UI).
* Mint ephemeral tokens with `mint_token(role_template, scope, ttl_minutes, reason, request_id)` by calling your GovernanceFactory wrapper (which uses long-lived credentials stored in Vault).
* Maintain an agent registry and token metadata (opaque IDs) — but avoid storing raw token material persisted to disk.
* Revoke tokens on request or on TTL expiry.
* Append and sign audit log entries (hash chain).

## 4. API surface (endpoints + examples)

This section defines the Controller REST API for Workers and Operators. Examples assume JSON over HTTPS and JWT or mTLS auth for clients.

### Authentication

* Operator: uses admin JWT or mTLS client cert.
* Worker: mutual TLS or short-lived client JWT (issued by Controller itself after Worker registers). Worker requests a token via `/mint-token`.

### Endpoints

#### POST /api/v1/mint-token

Mint an ephemeral token for a role template.

Request body:

```json
{
  "request_id":"uuid",
  "role_template":"repo_creator",
  "provider":"github",
  "scope":{"org":"acme"},
  "ttl_minutes":30,
  "reason":"bootstrap tenant acme-1",
  "parent_agent_id":null
}
```

Response:

```json
{
  "request_id":"uuid",
  "token_id":"opaque-id",
  "token":"<sensitive-token-material>",
  "expires_at":"ISO8601",
  "scopes":["repos:create","actions:write"]
}
```

> **Security note:** The Controller returns the token only over a secure, ephemeral websocket or an authenticated one-time retrieval endpoint. The raw token material must **not** be logged in cleartext.

---

#### POST /api/v1/revoke-token

Revoke an ephemeral token.

Request:

```json
{ "token_id":"opaque-id", "request_id":"uuid", "reason":"completed" }
```

Response: `{ "status":"ok" }`

---

#### POST /api/v1/plan-changes

Get a dry-run plan for a proposed change.

Request:

```json
{
  "request_id":"uuid",
  "provider":"github",
  "org":"acme",
  "plan": [
    {"op":"create_repo","name":"app-1"},
    {"op":"create_team","name":"app-1-devs"}

}
```

Response:

```json

  "request_id":"uuid",
  "plan": [ /* planned ops with estimated API calls */],

```

Operator endpoint for approving a previously plan-changes request.

Request:

```json

```

Response: `{ "status":"approved" }`

Workers may push events (Controller should provide an SDK function to sign & forward). Controller will append these events after validation.

Event body:

```json

  "request_id":"uuid",

```

Response: `{ "status":"ok","audit_id":"sha256..." }`

## 5. Data models

### TokenBlob (metadata stored by Controller)

```json
{
  "token_id":"opaque-id",
  "role_template":"repo_creator",
  "provider":"github",
  "scopes":["repos:create","actions:write"],
  "created_at":"ISO8601",

  "status":"active|revoked"
}
```

**Important:** Do not persist `token` (material) to persistent logs or DB. Only a one-time secure channel should transmit the token to the Worker. Token material should only be available in-memory during issuance.

### Agent registry entry

```json
{ "agent_id":"uuid","persona":"founder","spawned_at":"ISO8601","parent_agent":"uuid","status":"active" }
```

## 6. Token & credential flow (step-by-step)

1. Worker calls Controller `/mint-token` with `role_template` and `request_id`.
2. Controller validates policy, quotas, and whether `approval_required` is set.

   * If approval required: Controller writes a Plan and marks state `PENDING_APPROVAL` and notifies Operator.
3. If policy OK, Controller calls the appropriate GovernanceFactory wrapper method: `github_gov.mint_token(role_template, org, ttl_minutes, reason, request_id)`.

   * The factory returns `{id, token, expires_at, scopes}`.
4. Controller stores the `TokenBlob` metadata and returns the token to the Worker over a one-time retrieval endpoint or ephemeral websocket.
5. Worker uses token to call GovernanceFactory wrapper methods (factory library must accept that token to forward provider calls if your wrapper design requires it).
6. On success/failure, Worker posts audit events to `/api/v1/audit` and calls `/api/v1/revoke-token` if it's done.
7. Controller periodically scans for expired tokens and revokes them (cron job).

## 7. Audit log design (hash-chain + signing)

Each audit entry is appended to the audit store and then a `sha256` hash of the concatenation of the previous hash + entry JSON is computed and stored as `entry_hash`. The Controller signs the `entry_hash` with its private key to provide non-repudiation.

Sample audit entry JSON:

```json
{
  "audit_id":"sha256(entry_json)",
  "prev_hash":"sha256(prev_entry_json)",
  "entry":{...},
  "entry_hash":"sha256(prev_hash + entry_json)",
  "signature":"base64(signed(entry_hash, controller_priv_key))"
}
```

Store the audit entries in a WORM storage (S3 with object lock, append-only DB, or a ledger DB) and periodically publish a root hash to an external immutable anchor (e.g., blockchain/timestamping) if high-assurance is required.

## 8. Security posture & operational controls

* **Vault integration:** Controller reads long-lived provider creds from Vault on startup. Use short-lived Vault tokens and rotate them.
* **Network security:** Controller API only exposed internally; Worker requests go through a private network or mTLS gateway.
* **Authentication:** Operators use RBAC (JWT claims). Workers must register and present a cryptographic key or use mTLS.
* **Token leakage mitigation:** tokens are single-retrieval; token material never stored in DB or logs.
* **Rate-limits & throttles:** Controller enforces API call quotas per-agent and per-tenant.
* **Kill-switch:** Admin endpoint to `freeze_tenant` which revokes all active tokens for that tenant and sets state to `FROZEN`.

## 11. Minimal Flask controller example (code)

Below is a compact, runnable Flask example suitable for local POC. The real repo includes modular design, dependency injection, and better error handling.

```python
# controller_minimal.py
from flask import Flask, request, jsonify
import uuid, time
from mock_github_factory import MockGitHubFactory
from audit import AuditLog

app = Flask(__name__)

github_gov = MockGitHubFactory()  # local mock; in prod use GitHubGovernanceFactory(vault_creds)
audit = AuditLog()

@app.route('/api/v1/mint-token', methods=['POST'])
def mint_token():
    body = request.get_json()
    req_id = body.get('request_id') or str(uuid.uuid4())
    role = body['role_template']
    provider = body['provider']
    ttl = body.get('ttl_minutes', 30)

    # policy checks (simplified)
    if role == 'org_admin':
        return jsonify({'error':'approval_required'}), 403

    # call the governance factory to mint
    token_blob = github_gov.mint_token(role_template=role, org=body['scope']['org'], ttl_minutes=ttl, reason=body.get('reason',''))

    # record metadata in audit (do not log token)
    audit.append({
        'request_id':req_id,'actor':'controller','action':'mint_token','role':role,'provider':provider,'token_id':token_blob['id']
    })

    return jsonify({'request_id':req_id,'token_id':token_blob['id'],'token':token_blob['token'],'expires_at':token_blob['expires_at']})

@app.route('/api/v1/revoke-token', methods=['POST'])
def revoke_token():
    body = request.get_json()
    token_id = body['token_id']
    github_gov.revoke_token(token_id)
    audit.append({'request_id':body.get('request_id'),'actor':'controller','action':'revoke_token','token_id':token_id})
    return jsonify({'status':'ok'})

if __name__ == '__main__':
    app.run(port=8080)
```

import hashlib, json

def append_audit_entry(audit_store, prev_hash, entry_json, signing_key):
    entry_bytes = json.dumps(entry_json, sort_keys=True).encode()
    entry_hash = hashlib.sha256(prev_hash.encode() + entry_bytes).hexdigest()
    signature = sign(entry_hash, signing_key)
    audit_record = { 'prev_hash': prev_hash, 'entry': entry_json, 'entry_hash': entry_hash, 'signature': signature }
    audit_store.put(entry_hash, audit_record)
    return entry_hash
